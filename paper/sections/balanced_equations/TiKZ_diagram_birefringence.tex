\begin{tikzpicture}[
    node distance=1.8cm,
    box/.style={draw, rounded corners, align=center, padding=6pt},
    arrow/.style={-{Latex[length=3mm]}, thick}
]

% Probability density
\node[box] (rho) {Probability Density \\ $\rho(x)$};

% Raw gradient
\node[box, below=1.5cm of rho] (grad) {Probability Gradient \\ $\nabla \rho$};

% Birefringent split
\node[box, below left=1.5cm and 2.5cm of grad] (gradA) {Mode A Gradient \\ $\nabla\rho^{(A)}$};
\node[box, below right=1.5cm and 2.5cm of grad] (gradB) {Mode B Gradient \\ $\nabla\rho^{(B)}$};

% Force-channel gradients
\node[box, below=1.5cm of gradA] (forcesA) {Force-Channel Coupling \\ $\{\nabla_{(f)}\rho^{(A)}\}$};
\node[box, below=1.5cm of gradB] (forcesB) {Force-Channel Coupling \\ $\{\nabla_{(f)}\rho^{(B)}\}$};

% Fluxes
\node[box, below=1.8cm of forcesA] (fluxA) {Flux (Mode A) \\ $\mathbf{J}^{(A)}$};
\node[box, below=1.8cm of forcesB] (fluxB) {Flux (Mode B) \\ $\mathbf{J}^{(B)}$};

% Transfer
\node[box, below=2cm of grad] (transfer) {Probability Transfer \\ $\Pi_{i\to j}$};

% Entanglement
\node[box, below=2cm of transfer] (ent) {Entanglement Channel \\ $\mathcal{E}_{i,j}$};

% Creation / Loss / Transfer
\node[box, below left=1.5cm and 2.5cm of ent] (create) {Creation \\ $\mathcal{C}_{i,j}$};
\node[box, below=1.5cm of ent] (loss) {Loss / Emission \\ $\mathcal{D}_{i,j}$};
\node[box, below right=1.5cm and 2.5cm of ent] (etrans) {Transfer \\ $\Pi_{k\to(i,j)}$};

% Arrows
\draw[arrow] (rho) -- (grad);

% Birefringent split
\draw[arrow] (grad) -- (gradA);
\draw[arrow] (grad) -- (gradB);

% Force-channel coupling
\draw[arrow] (gradA) -- (forcesA);
\draw[arrow] (gradB) -- (forcesB);

% Flux
\draw[arrow] (forcesA) -- (fluxA);
\draw[arrow] (forcesB) -- (fluxB);

% Flux recombination into transfer
\draw[arrow] (fluxA) |- (transfer);
\draw[arrow] (fluxB) |- (transfer);

% Transfer to entanglement
\draw[arrow] (transfer) -- (ent);

% Entanglement components
\draw[arrow] (ent) -- (create);
\draw[arrow] (ent) -- (loss);
\draw[arrow] (ent) -- (etrans);

\end{tikzpicture}
